## Daily JS

### Functions and Fat Arrow Syntax

```js
var x = function() {
  console.log('hello');
  return 'Passed from x';
};

function y() {
  console.log('world');
}

var z = () => {
  console.log('yup');
};

var a = () => {
  return 'no';
};


console.log(x); // [Function]
x(); // "hello"
console.log(x()); // "hello" // "Passed from x"
console.log(y()); // "world" // undefined
console.log(z()); // "yup" // undefined
console.log(a()); // "no"
```
